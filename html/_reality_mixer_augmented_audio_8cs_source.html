68  public static double Distance (Location p1, Location p2)
69  {
70  double R = 6371.0; // earth mean radius
71  double t1 = Degree2Radian (p1.Latitude);
72  double t2 = Degree2Radian (p2.Latitude);
73  double dt = Degree2Radian (p2.Latitude - p1.Latitude);
74  double ds = Degree2Radian (p2.Longitude - p1.Longitude);
75  double a = Math.Sin (dt / 2.0) * Math.Sin (dt / 2.0) + Math.Cos (t1) * Math.Cos (t2) * Math.Sin (ds / 2.0) * Math.Sin (ds / 2.0);
76  double c = 2.0 * Math.Atan2(Math.Sqrt(a), Math.Sqrt(1.0 - a));
77  return R * c;
78  }

196  public int PersistentIdFromPoI(Location loc, int modulus)
197  {
198  // get stable Gnome's voice Id from nearest PoI
199 
200  // approximate based on limits of Haversine function (see Location class of POI enabler)
201  double earthRadiusKM = 6378.137;
202  double toMetres = earthRadiusKM * 2.0 * (double)Mathf.PI * 1000.0f;
203 
204  // Lat Long to a form of geohash (0-180range)
205 
206  UInt64 geohash = (UInt64)(toMetres*((loc.Latitude+90.0)*180.0+(loc.Longitude+90.0)));
207  return (int)(geohash % (UInt64)modulus);

218  public double GPSBearing(double lat1, double lng1, double lat2, double lng2)
219  {
220  lat1*=(double)Mathf.Deg2Rad;
221  lng1*=(double)Mathf.Deg2Rad;
222  lat2*=(double)Mathf.Deg2Rad;
223  lng2*=(double)Mathf.Deg2Rad;
224  // approximate spherical trig bearing
225  double dLon = (lng1-lng2);
226  double y = Math.Sin(dLon)*Math.Cos(lat1);
227  double x = Math.Cos(lat2)*Math.Sin(lat1) - Math.Sin(lat2)*Math.Cos(lat1)*Math.Cos(dLon);
228  double bearing = Math.Atan2(y, x);
229  return bearing;
230  }

238  public double RelativeHeading(double hd1, double hd2)
239  {
240  double PI2=Math.PI*2.0;
241  if (hd1>PI2 || hd1<0.0 || hd2>PI2 || hd2<0.0)
242  return 0.0;
243 
244  double diff = hd2-hd1;
245  double absDiff = Math.Abs(diff);
246 
247  if (absDiff<=Math.PI)
248  return absDiff == Math.PI ? absDiff : diff;
249  else if (hd2>hd1)
250  return absDiff-PI2;
251 
252  return PI2-absDiff;
253  }

270  public void ProcessAudioSource()

284  {
285  int categoryId = PersistentIdFromPoI(poiLocation, maxCategories);
286  int vocal = UnityEngine.Random.Range(0,maxVocals);
287  if (vocal==lastVocal) vocal=(vocal+1) % maxVocals;
288 
289  theSource.clip=(vocalArray[categoryId*maxVocals+vocal]);

298 #endif
299

302 
303  #if UNITY_EDITOR
304  dist*=AUDIO_DISTANCE_SCALE;
305  #endif
306  if (dist<AUDIO_CULLING_DISTANCE)

316
